\section{Question GATE AG 14}
$y=e^{mx}+e^{-mx}$ is the solution of which differential equation?
\section{Solution}

\begin{equation}
    y = e^{mx} + e^{-mx} \label{eq:eq1}
\end{equation}

\begin{equation}
    \frac{dy}{dx} = m e^{mx} - m e^{-mx}
\end{equation}

\begin{equation}
    \frac{d^{2}y}{dx^{2}} = m^2 e^{mx} + m^2 e^{-mx}
\end{equation}

\begin{equation}
    \frac{d^{2}y}{dx^{2}} = m^2(e^{mx} + e^{-mx})
\end{equation}
 Using value of y from \ref{eq:eq1}
\begin{equation}
    \frac{d^{2}y}{dx^{2}} - m^2y = 0
\end{equation}
